<script>
    const cells = document.querySelectorAll('.cell');
    const statusText = document.getElementById('status');
    const pvpBtn = document.getElementById('pvpBtn');
    const pveBtn = document.getElementById('pveBtn');
    const langToggle = document.getElementById('langToggle');
    const gameTitle = document.getElementById('gameTitle');
    const resetBtn = document.getElementById('resetBtn');

    // Các biến điều khiển của Modal
    const endGameModal = document.getElementById('endGameModal');
    const modalEmoji = document.getElementById('modalEmoji');
    const modalText = document.getElementById('modalText');
    const modalResetBtn = document.getElementById('modalResetBtn');

    let gameMode = 'pvp'; 
    let currentPlayer = 'X'; 
    let gameState = ["", "", "", "", "", "", "", "", ""];
    let isGameActive = true;
    let currentLang = 'vi'; 

    // Bộ từ điển dịch đa ngôn ngữ nâng cấp
    const translations = {
        vi: {
            title: "Tic-Tak-Toe 3x3",
            pvp: "👥 Chơi 2 người",
            pve: "🤖 Đấu với Máy",
            reset: "Chơi lại 🔄",
            turnX: "Lượt của X",
            turnXYou: "Lượt của X (Bạn)",
            turnO: "Lượt của O",
            botThinking: "Máy đang tính toán...",
            win: "Người chơi {player} đã CHIẾN THẮNG! 🎉",
            botWin: "Máy đã thắng! Bạn không thể thắng được đâu 🤖",
            draw: "Trò chơi HÒA! 🤝",
            modalReset: "Chơi lại 🔄"
        },
        en: {
            title: "Tic-Tak-Toe 3x3",
            pvp: "👥 2 Players",
            pve: "🤖 Vs Computer",
            reset: "Reset Game 🔄",
            turnX: "X's Turn",
            turnXYou: "X's Turn (You)",
            turnO: "O's Turn",
            botThinking: "Computer thinking...",
            win: "Player {player} WINS! 🎉",
            botWin: "Computer wins! AI is unbeatable 🤖",
            draw: "It's a DRAW! 🤝",
            modalReset: "Play Again 🔄"
        }
    };

    const winningConditions = [
        [0, 1, 2], [3, 4, 5], [6, 7, 8],
        [0, 3, 6], [1, 4, 7], [2, 5, 8],
        [0, 4, 8], [2, 4, 6]
    ];

    function toggleLanguage() {
        currentLang = currentLang === 'vi' ? 'en' : 'vi';
        langToggle.textContent = currentLang === 'vi' ? '🌐 English' : '🌐 Tiếng Việt';
        
        gameTitle.textContent = translations[currentLang].title;
        pvpBtn.textContent = translations[currentLang].pvp;
        pveBtn.textContent = translations[currentLang].pve;
        resetBtn.textContent = translations[currentLang].reset;
        modalResetBtn.textContent = translations[currentLang].modalReset;

        updateStatusDisplay();
    }

    function updateStatusDisplay(endStatus = null) {
        if (endStatus) {
            statusText.textContent = endStatus;
            return;
        }

        if (!isGameActive) return;

        if (currentPlayer === 'X') {
            statusText.textContent = gameMode === 'pve' 
                ? translations[currentLang].turnXYou 
                : translations[currentLang].turnX;
        } else {
            statusText.textContent = gameMode === 'pve'
                ? translations[currentLang].botThinking
                : translations[currentLang].turnO;
        }
    }

    // Thay đổi chế độ chơi
    function setMode(mode) {
        gameMode = mode;
        if (mode === 'pvp') {
            pvpBtn.classList.add('active');
            pveBtn.classList.remove('active');
        } else {
            pveBtn.classList.add('active');
            pvpBtn.classList.remove('active');
        }
        resetGame();
    }

    function handleCellClick(e) {
        const clickedCell = e.target;
        const clickedCellIndex = parseInt(clickedCell.getAttribute('data-cell-index'));

        if (gameState[clickedCellIndex] !== "" || !isGameActive) {
            return;
        }

        makeMove(clickedCellIndex, currentPlayer);

        if (!isGameActive) return;

        if (gameMode === 'pve') {
            currentPlayer = 'O';
            updateStatusDisplay();
            
            setTimeout(() => {
                const bestMove = findBestMove(gameState);
                makeMove(bestMove, 'O');
                if (isGameActive) {
                    currentPlayer = 'X';
                    updateStatusDisplay();
                }
            }, 400);
        } else {
            currentPlayer = currentPlayer === 'X' ? 'O' : 'X';
            updateStatusDisplay();
        }
    }

    function makeMove(index, player) {
        gameState[index] = player;
        cells[index].textContent = player;
        cells[index].classList.add(player.toLowerCase());
        checkResult();
    }

    // Hàm kích hoạt hiệu ứng toàn màn hình
    function triggerEndGameEffect(type, message) {
        endGameModal.className = `modal-overlay ${type}-effect`;
        modalText.textContent = message;

        if (type === 'win') {
            modalEmoji.textContent = "🏆";
        } else if (type === 'lose') {
            modalEmoji.textContent = "👾";
        } else {
            modalEmoji.textContent = "🤝";
        }

        endGameModal.style.display = 'flex';
    }

    // Đóng màn hình hiệu ứng và reset
    function closeModalAndReset() {
        endGameModal.style.display = 'none';
        resetGame();
    }

    function checkResult() {
        let roundWon = false;
        for (let i = 0; i < winningConditions.length; i++) {
            const [a, b, c] = winningConditions[i];
            if (gameState[a] && gameState[a] === gameState[b] && gameState[a] === gameState[c]) {
                roundWon = true;
                break;
            }
        }

        if (roundWon) {
            let endMsg = "";
            let effectType = "win";

            if (gameMode === 'pve' && currentPlayer === 'O') {
                endMsg = translations[currentLang].botWin;
                effectType = "lose"; 
            } else {
                endMsg = translations[currentLang].win.replace("{player}", currentPlayer);
                effectType = "win"; 
            }

            updateStatusDisplay(endMsg);
            isGameActive = false;

            setTimeout(() => {
                triggerEndGameEffect(effectType, endMsg);
            }, 500);
            return;
        }

        if (!gameState.includes("")) {
            const drawMsg = translations[currentLang].draw;
            updateStatusDisplay(drawMsg);
            isGameActive = false;

            setTimeout(() => {
                triggerEndGameEffect('draw', drawMsg);
            }, 500);
        }
    }

    function resetGame() {
        currentPlayer = 'X';
        gameState = ["", "", "", "", "", "", "", "", ""];
        isGameActive = true;
        updateStatusDisplay();
        cells.forEach(cell => {
            cell.textContent = "";
            cell.className = "cell";
        });
    }

    cells.forEach(cell => cell.addEventListener('click', handleCellClick));

    /* ================= THUẬT TOÁN MINIMAX AI (BẤT BẠI) ================= */
    function evaluateBoard(board) {
        for (let i = 0; i < winningConditions.length; i++) {
            const [a, b, c] = winningConditions[i];
            if (board[a] === board[b] && board[b] === board[c]) {
                if (board[a] === 'O') return 10;
                if (board[a] === 'X') return -10;
            }
        }
        return 0;
    }

    function isMovesLeft(board) {
        return board.includes("");
    }

    function minimax(board, depth, isMax) {
        let score = evaluateBoard(board);

        if (score === 10) return score - depth;
        if (score === -10) return score + depth;
        if (!isMovesLeft(board)) return 0;

        if (isMax) {
            let best = -Infinity;
            for (let i = 0; i < 9; i++) {
                if (board[i] === "") {
                    board[i] = 'O';
                    best = Math.max(best, minimax(board, depth + 1, false));
                    board[i] = "";
                }
            }
            return best;
        } else {
            let best = Infinity;
            for (let i = 0; i < 9; i++) {
                if (board[i] === "") {
                    board[i] = 'X';
                    best = Math.min(best, minimax(board, depth + 1, true));
                    board[i] = "";
                }
            }
            return best;
        }
    }

    function findBestMove(board) {
        let bestVal = -Infinity;
        let bestMove = -1;

        for (let i = 0; i < 9; i++) {
            if (board[i] === "") {
                board[i] = 'O';
                let moveVal = minimax(board, 0, false);
                board[i] = "";

                if (moveVal > bestVal) {
                    bestMove = i;
                    bestVal = moveVal;
                }
            }
        }
        return bestMove;
    }
