        // Hàm cập nhật ngôn ngữ giao diện
        function updateLanguage() {
            const t = homeTranslations[currentLang];
            
            // Cập nhật thẻ tiêu đề trình duyệt
            document.getElementById('docTitle').textContent = t.docTitle;
            
            // Cập nhật thanh menu & nút chuyển ngôn ngữ
            document.getElementById('nav-about').textContent = t.navAbout;
            document.getElementById('nav-games').textContent = t.navGames;
            document.getElementById('langToggle').textContent = t.langBtn;
            
            // Cập nhật trang Giới thiệu
            document.getElementById('aboutTitle').textContent = t.aboutTitle;
            document.getElementById('aboutSubtitle').textContent = t.aboutSubtitle;
            document.getElementById('aboutPlaceholder').innerHTML = `
                <h3>${t.aboutPlaceholderTitle}</h3>
                <p style="margin-top: 10px;">${t.aboutPlaceholderDesc}</p>
            `;
            
            // Cập nhật trang Kho Game
            document.getElementById('gamesTitle').textContent = t.gamesTitle;
            document.getElementById('gamesSubtitle').textContent = t.gamesSubtitle;
            
            // Game 1
            document.getElementById('game1Title').textContent = t.game1Title;
            document.getElementById('game1Desc').textContent = t.game1Desc;
            document.getElementById('game1Btn').textContent = t.game1Btn;
            
            // Footer
            document.getElementById('footerText').textContent = t.footerText;
        }

        // Sự kiện đổi ngôn ngữ khi nhấn nút
        function toggleLanguage() {
            currentLang = currentLang === 'vi' ? 'en' : 'vi';
            updateLanguage();
        }

        // Xử lý chuyển đổi giữa các trang con (Single Page App)
        function switchPage(pageId) {
            event.preventDefault();
            currentPage = pageId;

            // Ẩn tất cả các sections
            document.querySelectorAll('.page-section').forEach(section => {
                section.classList.remove('active');
            });

            // Gỡ class active của menu điều hướng
            document.querySelectorAll('.nav-links a').forEach(link => {
                link.classList.remove('active');
            });

            // Hiển thị phần nội dung được chọn
            if (pageId === 'about') {
                document.getElementById('about-section').classList.add('active');
                document.getElementById('nav-about').classList.add('active');
            } else if (pageId === 'games') {
                document.getElementById('games-section').classList.add('active');
                document.getElementById('nav-games').classList.add('active');
            }
        }

        // Khởi chạy ngôn ngữ mặc định (Tiếng Việt) khi tải trang
        updateLanguage();
